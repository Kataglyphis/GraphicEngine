108void ShaderProgram::add_shader(GLuint program, const char* shader_code, GLenum shader_type)
109{
110 GLuint shader = glCreateShader(shader_type);
111
112 // the opengl function wants c -style char array of code and the length in an array ... so we do it
113 const GLchar* code[1];
114 code[0] = shader_code;
115
116 GLint code_length[1];
117 code_length[0] = (GLint)strlen(shader_code);
118
119 glShaderSource(shader, 1, code, code_length);
120 glCompileShader(shader);
121 //glCompileShaderIncludeARB(shader);
122
123 GLint result = 0;
124 GLchar eLog[1024] = { 0 };
125
126 //retrieve status of the shader and print if any error occured
127 glGetShaderiv(shader, GL_COMPILE_STATUS, &result);
128
129 if (!result) {
130 glGetShaderInfoLog(shader, sizeof(eLog), NULL, eLog);
131 printf("Error compiling the %d shader: '%s'\n", shader_type, eLog);
132 printf("%s", shader_code);
133 return;
134 }
135
136 // we are happy, everything went well so attach shader to program
137 glAttachShader(program, shader);
138}
