116void DirectionalLight::calc_orthogonal_projections(
117 glm::mat4 camera_view_matrix, GLfloat fov, GLuint window_width,
118 GLuint window_height, GLuint current_num_cascades) {
119 // calc the start and end point for our cascaded shadow maps
120 calc_cascaded_slots();
121
122 for (int i = 0; i < static_cast<int>(current_num_cascades); i++) {
123 glm::mat4 curr_cascade_proj = glm::perspective(
124 glm::radians(fov), (float)window_width / (float)window_height,
125 cascade_slots[i], cascade_slots[i + 1]);
126
127 std::vector<glm::vec4> frustumCornerWorldSpace =
128 get_frustum_corners_world_space(curr_cascade_proj, camera_view_matrix);
129
130 glm::vec3 center = glm::vec3(0, 0, 0);
131 for (const auto& v : frustumCornerWorldSpace) {
132 center += glm::vec3(v);
133 }
134
135 center /= frustumCornerWorldSpace.size();
136
137 glm::mat4 light_view_matrix = glm::lookAt(center - get_direction(), center,
138 glm::vec3(0.0f, 1.0f, 0.0f));
139
140 // the # of frustum corners = 8
141 GLfloat minX = std::numeric_limits<float>::max();
142 GLfloat maxX = std::numeric_limits<float>::min();
143 GLfloat minY = std::numeric_limits<float>::max();
144 GLfloat maxY = std::numeric_limits<float>::min();
145 GLfloat minZ = std::numeric_limits<float>::max();
146 GLfloat maxZ = std::numeric_limits<float>::min();
147
148 for (unsigned int m = 0; m < frustumCornerWorldSpace.size(); m++) {
149 // transform each corner from view to world space
150 glm::vec4 v_light_view = light_view_matrix * frustumCornerWorldSpace[m];
151 // now go to light space
152 minX = std::min(minX, v_light_view.x);
153 maxX = std::max(maxX, v_light_view.x);
154 // we always have negativ y values
155 minY = std::min(minY, v_light_view.y);
156 maxY = std::max(maxY, v_light_view.y);
157
158 minZ = std::min(minZ, v_light_view.z);
159 maxZ = std::max(maxZ, v_light_view.z);
160 }
161
162 // Tune this parameter according to the scene
163 // for having objects casting shadows that are actually not in the frustum
164 // :)
165 constexpr float zMult = 10.0f;
166 if (minZ < 0) {
167 minZ *= zMult;
168 } else {
169 minZ /= zMult;
170 }
171 if (maxZ < 0) {
172 maxZ /= zMult;
173 } else {
174 maxZ *= zMult;
175 }
176
177 glm::mat4 light_projection = glm::ortho(minX, maxX, minY, maxY, minZ, maxZ);
178
179 cascade_light_matrices[i] = light_projection * light_view_matrix;
180 }
181}
